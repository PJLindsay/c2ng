# Grammar for member functions

member-variable = 'm_' <lc> property-name
                | 'm_p' property-name
                | 'm_' object-accessor

member-function = 'get' property-name | 'get' | 'get' property-name 'From' property-name | 'get' property-name 'By' property-name
                | 'tryGet' property-name
                | 'set' property-name | 'set'
                | 'store' property-name
                | 'parse' property-name
                | 'fire' property-name
                | 'start' property-name
                | 'kill' property-name
                | 'land' property-name
                | 'add' property-name | 'add'
                | 'addTo' property-name
                | 'change' property-name | 'change'
                | 'enumerate' property-name
                | 'delete' property-name
                | 'update' property-name | 'update'
                | 'compute' property-name
                | 'sort' property-name | 'sort' | 'sortForNewShip'
                | 'load' property-name | 'load'
                | 'to' conversion-target
                | 'from' conversion-target
                | 'is' testable-property
                | 'has' property-name
                | 'clear' property-name | 'clear'
                | 'remove' property-name | 'remove'
                | random-member
                | object-accessor
